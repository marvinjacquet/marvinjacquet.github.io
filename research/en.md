<script>
  const buttons = document.querySelectorAll('.filter-btn');
  const cards = document.querySelectorAll('.item-card');
  const groups = document.querySelectorAll('.section-group');

  buttons.forEach(btn => {
    btn.addEventListener('click', () => {
      buttons.forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      const filter = btn.dataset.filter;
      cards.forEach(card => {
        card.classList.toggle('hidden', filter !== 'all' && card.dataset.type !== filter);
      });
      groups.forEach(group => {
        const visible = group.querySelectorAll('.item-card:not(.hidden)');
        group.classList.toggle('hidden', visible.length === 0);
      });
    });
  });
</script>
